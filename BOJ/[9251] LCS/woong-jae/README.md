    if(index1 == str1.size() || index2 == str2.size()) return 0;
    int&ret = cache[index1][index2];
    if(ret != -1) return ret;
    
    ret = 0;
    if(str1[index1] == str2[index2]) {
        ret = max(ret, lcs(index1 + 1, index2 + 1) + 1);
    } else {
        ret = max(lcs(index1 + 1, index2), lcs(index1, index2 + 1));
    }
    
    return ret;
}
